Hello CLI Chat Widget

Demonstração interativa do componente de chat para aplicações web e mobile via WebView

Como implementar

// 1. Inclua os arquivos em seu projeto
<link rel="stylesheet" href="styles.css">
<script src="index.js"></script>

// 2. Adicione um elemento container
<div id="root"></div>

// 3. Inicialize o widget
<script>
  const config = {
    title: "Chat Assistente",
    primaryColor: "#3b82f6",
    initialMessage: "Olá! Como posso ajudar?",
    position: "right"
  };
  
  window.addEventListener('load', function() {
    HelloCLI.renderChatWidget('root', config);
  });
</script>
